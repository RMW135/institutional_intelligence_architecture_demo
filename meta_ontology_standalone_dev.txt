//   layer. Includes its own mock company graph so it can be loaded into a
//   separate Neo4j instance without requiring the Full Intelligence System
//   seed graph.
//
// REVISION NOTES (2026-04-12):
//   - Confidence scores REMOVED from LogicalRelation nodes.
//     Rationale: A logical relation (material implication, sequence, etc.)
//     is either valid or not. It carries no intrinsic uncertainty.
//     Uncertainty belongs on the grounding step — the evidence that caused
//     a proposition to be asserted — not on the relation itself.
//     This was a category error in the prior version.
//
//   - ConnectiveType nodes for AND, OR, NOT, IMP, IFF, XOR are now
//     explicitly marked as DERIVED ALIASES. They are not primitives.
//     All propositional connectives are constructible from NAND + negated
//     endpoint flags (see fol_primitive_substrate.cypher).
//     These nodes are retained for human legibility only.
//
//   - Behavioral profile operators (CAUSAL, SEQ, EXCL, COEX, RESP) are
//     retained as a SEPARATE LAYER. These are process-level operators
//     from van der Aalst et al. (2011). They do NOT reduce to NAND —
//     they operate over event traces, not propositions. They belong to
//     a distinct ontological layer and are marked accordingly.
//
//   - LogicalRelation nodes now carry a verified:boolean property only.
//     Removed: confidence (category error as noted above).
//
// STRUCTURE:
//   Part A: Mock Company Graph
//   Part B: Derived Connective Aliases (human legibility layer)
//   Part C: Process Operator Layer (van der Aalst behavioral operators)
//   Part D: Propositions
//   Part E: Logical Relations (reified, confidence-free)
//   Part F: Quantifier Nodes + GROUNDS_IN Bridges
//   Part G: Drift Detection + Hypothesis Generation
//   Part H: Verification and Development Queries

// PART B: DERIVED CONNECTIVE ALIASES
// ═══════════════════════════════════════════════════════════════════════════════
//
// These ConnectiveType nodes are DERIVED ALIASES only — human-legible names
// for specific NAND compositions. They are NOT logical primitives.
//
// The actual primitive substrate is defined in fol_primitive_substrate.cypher:
//   Primitive 1: NAND edge (sole primitive relation type)
//   Primitive 2: negated endpoint flags (start_negated, end_negated on edge)
//
// NAND decomposition of each alias:
//   AND(A,B)   = NOT(NAND(A,B))     — pure NAND with negated result context
//   OR(A,B)    = NAND(NOT(A),NOT(B)) — start_negated:true, end_negated:true
//   NOT(A)     = A with negated endpoint flag = true (no separate edge needed)
//   IMP(A,B)   = NAND(A,NOT(B))     — start_negated:false, end_negated:true
//   IFF(A,B)   = AND(IMP(A,B),IMP(B,A)) — two NAND compositions
//   XOR(A,B)   = AND(OR(A,B),NAND(A,B)) — three NAND compositions
//
// These aliases are retained because:
//   1. Human legibility for ontologists without FOL background
//   2. The Esser/Fahland reification mechanism references named connectives
//   3. The academic paper requires named connective types for formalization

MERGE (ct:ConnectiveType {symbol: 'NAND'})
ON CREATE SET
  ct.latex        = '\\uparrow',
  ct.arity        = 2,
  ct.primitive    = true,
  ct.derived_from = 'none',
  ct.commutative  = true,
  ct.associative  = false,
  ct.description  = 'Sheffer stroke — the sole logical primitive. NOT(A AND B). Logically complete alone.'
ON MATCH SET
  ct.primitive    = true,
  ct.derived_from = 'none';

MERGE (:ConnectiveType {
  symbol:       'AND',
  latex:        '\\wedge',
  arity:        2,
  primitive:    false,
  derived_from: 'NOT(NAND(A,B))',
  commutative:  true,
  associative:  true,
  description:  'Conjunction [DERIVED]. Both operands hold. = NOT(NAND(A,B)).'
});

MERGE (:ConnectiveType {
  symbol:       'OR',
  latex:        '\\vee',
  arity:        2,
  primitive:    false,
  derived_from: 'NAND(NOT(A),NOT(B))',
  commutative:  true,
  associative:  true,
  description:  'Disjunction [DERIVED]. At least one operand holds. = NAND(NOT(A),NOT(B)).'
});

MERGE (:ConnectiveType {
  symbol:       'NOT',
  latex:        '\\neg',
  arity:        1,
  primitive:    false,
  derived_from: 'endpoint negated flag on NAND edge',
  commutative:  false,
  associative:  false,
  description:  'Negation [DERIVED]. Encoded as start_negated or end_negated property on NAND edge. No separate edge needed.'
});

MERGE (:ConnectiveType {
  symbol:       'IMP',
  latex:        '\\rightarrow',
  arity:        2,
  primitive:    false,
  derived_from: 'NAND(A,NOT(B))',
  commutative:  false,
  associative:  false,
  description:  'Material implication [DERIVED]. If antecedent holds then consequent holds. = NAND(A,NOT(B)) = start_negated:false, end_negated:true.'
});

MERGE (:ConnectiveType {
  symbol:       'IFF',
  latex:        '\\leftrightarrow',
  arity:        2,
  primitive:    false,
  derived_from: 'AND(IMP(A,B),IMP(B,A))',
  commutative:  true,
  associative:  false,
  description:  'Biconditional [DERIVED]. Both hold or neither holds. = AND(IMP(A,B),IMP(B,A)). Requires two NAND compositions.'
});

MERGE (:ConnectiveType {
  symbol:       'XOR',
  latex:        '\\oplus',
  arity:        2,
  primitive:    false,
  derived_from: 'AND(OR(A,B),NAND(A,B))',
  commutative:  true,
  associative:  false,
  description:  'Exclusive disjunction [DERIVED]. Exactly one operand holds. = AND(OR(A,B),NAND(A,B)). Requires three NAND compositions.'
});

// Behavioral profile operators from van der Aalst, Adriansyah & de Medeiros (2011).

//

// PART E: LOGICAL RELATIONS (REIFIED, CONFIDENCE-FREE)
// ═══════════════════════════════════════════════════════════════════════════════
//
// LogicalRelation nodes reify the connections between Propositions.
// They instantiate a ConnectiveType and carry antecedent/consequent edges.
//
// REMOVED: confidence property.
// RATIONALE: A logical relation is either valid or not. Validity is a
// structural property of the relation's grounding in the institutional
// ontology — not a probabilistic assessment. Uncertainty belongs on the
// grounding step (what evidence caused a Proposition to be asserted),
// not on the logical relation itself. Confidence on a material implication
// is a category error — it conflates epistemic uncertainty about premises
// with the logical form of the inference.
//
// Schema:
//   id        — unique
//   label     — human-readable name
//   source    — what authored this relation ('ArchitecturalInvariant', etc.)
//   verified  — whether a knowledge architect has validated this relation
//   created   — date

// PART G: DRIFT DETECTION + HYPOTHESIS GENERATION
// ═══════════════════════════════════════════════════════════════════════════════
//
// EXPECTED RESULTS with mock data:
//
//   MockExpensePolicy (mock_POL1):
//     D001 (Alice/MockAnalyst, escalated) + D002 (Bob/MockManager, escalated)
//     + D003 (Alice/MockAnalyst, escalated) = 3 decisions, 2 unique roles
//     Q001: freq >= 2 ✓, roles >= 2 ✓ → DRIFT DETECTED
//     All statuses 'escalated' → type: RelationshipMisdrawn
//
//   MockAuditAccessPolicy (mock_POL2):
//     D005 (Bob/MockManager, denied) + D006 (Alice/MockAnalyst, denied)
//     = 2 decisions, 2 unique roles
//     Q002: freq >= 2 ✓, roles >= 2 ✓ → DRIFT DETECTED
//     Statuses include 'denied' → type: EntityReclassification

MATCH (q:Quantifier)-[:GROUNDS_IN]->(pol:Policy)
      <-[:TRIGGERED_POLICY]-(d:Decision)
      -[:MADE_BY]->(person:Person)
      -[:HAS_ROLE]->(role:Role)
WHERE d.status IN ['escalated', 'denied']
WITH q, pol,
     collect(DISTINCT role.name)  AS roles_involved,
     count(DISTINCT role.name)    AS unique_roles,
     count(DISTINCT d)            AS frequency,
     collect(DISTINCT d.status)   AS statuses
WHERE unique_roles >= q.threshold_unique_roles
  AND frequency   >= q.threshold_frequency
WITH q, pol, roles_involved, unique_roles, frequency, statuses,
     CASE
       WHEN size([s IN statuses WHERE s = 'denied']) > 0
         THEN 'EntityReclassification'
       WHEN size([s IN statuses WHERE s = 'escalated']) = size(statuses)
         THEN 'RelationshipMisdrawn'
       ELSE 'OntologyGap'
     END AS hyp_type
MERGE (dh:DriftHypothesis {
  id: 'DH_' + q.id + '_' + pol.policy_id
})
ON CREATE SET
  dh.type               = hyp_type,
  dh.label              = hyp_type + '_' + pol.name,
  dh.description        = 'Drift detected: ' + toString(frequency) +
                          ' decisions across ' + toString(unique_roles) + ' roles',
  dh.frequency_observed = frequency,
  dh.unique_roles       = unique_roles,
  dh.roles_involved     = roles_involved,
  dh.status             = 'candidate',
  dh.created            = datetime()
ON MATCH SET
  dh.frequency_observed = frequency,
  dh.unique_roles       = unique_roles,
  dh.roles_involved     = roles_involved,
  dh.last_evaluated     = datetime()
MERGE (dh)-[:CONCERNS]->(pol)
MERGE (dh)-[:GENERATED_BY]->(q);

// H1. Confirm connective type taxonomy with primitive/derived distinction
MATCH (ct:ConnectiveType)
RETURN ct.symbol AS symbol, ct.primitive AS is_primitive,
       ct.derived_from AS derived_from,
       ct.layer AS layer,
       ct.description AS description
ORDER BY ct.primitive DESC, ct.symbol;

// H2. Show all decisions grouped by policy
MATCH (d:Decision)-[:TRIGGERED_POLICY]->(pol:Policy)
OPTIONAL MATCH (d)-[:MADE_BY]->(person:Person)-[:HAS_ROLE]->(role:Role)
RETURN pol.name AS policy, d.decision_id AS decision,
       d.status AS status, person.name AS actor, role.name AS role
ORDER BY pol.name, d.timestamp;

// H3. Quantifier threshold vs actual counts
MATCH (q:Quantifier)-[:GROUNDS_IN]->(pol:Policy)
      <-[:TRIGGERED_POLICY]-(d:Decision)
      -[:MADE_BY]->(person:Person)
      -[:HAS_ROLE]->(role:Role)
WHERE d.status IN ['escalated', 'denied']
WITH q, pol,
     count(DISTINCT role.name) AS actual_unique_roles,
     count(DISTINCT d)         AS actual_frequency
RETURN
  q.id                      AS quantifier,
  pol.name                  AS policy,
  q.threshold_frequency     AS freq_threshold,
  actual_frequency          AS freq_actual,
  actual_frequency >= q.threshold_frequency AS freq_met,
  q.threshold_unique_roles  AS role_threshold,
  actual_unique_roles       AS roles_actual,
  actual_unique_roles >= q.threshold_unique_roles AS roles_met;

// H4. Generated DriftHypothesis nodes
MATCH (dh:DriftHypothesis)-[:CONCERNS]->(pol:Policy)
OPTIONAL MATCH (dh)-[:GENERATED_BY]->(q:Quantifier)
RETURN
  dh.id                 AS hypothesis_id,
  dh.type               AS hypothesis_type,
  dh.frequency_observed AS observed_frequency,
  dh.unique_roles       AS observed_unique_roles,
  dh.roles_involved     AS roles,
  pol.name              AS policy_concerned,
  q.id                  AS triggered_by;

// H5. Full three-layer trace: LogicalRelation → Quantifier → Policy → Decisions
MATCH (lr:LogicalRelation)-[:HAS_ANTECEDENT]->(q:Quantifier)
      -[:GROUNDS_IN]->(pol:Policy)
      <-[:TRIGGERED_POLICY]-(d:Decision)
      -[:MADE_BY]->(person:Person)
      -[:HAS_ROLE]->(role:Role)
WHERE d.status IN ['escalated', 'denied']
RETURN
  lr.label       AS drift_rule,
  q.id           AS quantifier,
  pol.name       AS policy,
  d.decision_id  AS decision,
  d.status       AS status,
  person.name    AS actor,
  role.name      AS actor_role
ORDER BY pol.name, d.timestamp;

// H6. Verify no LogicalRelation nodes carry confidence (category error check)
MATCH (lr:LogicalRelation)
WHERE lr.confidence IS NOT NULL
RETURN count(lr) AS logical_relations_with_confidence_SHOULD_BE_ZERO;

// H7. Show logical relation chain with connective types (no confidence column)
MATCH (lr:LogicalRelation)-[:INSTANTIATES]->(ct:ConnectiveType)
OPTIONAL MATCH (lr)-[:HAS_ANTECEDENT]->(ant)
OPTIONAL MATCH (lr)-[:HAS_CONSEQUENT]->(con)
RETURN
  lr.id        AS relation_id,
  lr.label     AS label,
  ct.symbol    AS connective,
  ct.primitive AS connective_is_primitive,
  lr.source    AS source,
  lr.verified  AS verified,
  labels(ant)[0] AS antecedent_type,
  COALESCE(ant.label, ant.id) AS antecedent,
  labels(con)[0] AS consequent_type,
  COALESCE(con.label, con.id) AS consequent
ORDER BY lr.id;
